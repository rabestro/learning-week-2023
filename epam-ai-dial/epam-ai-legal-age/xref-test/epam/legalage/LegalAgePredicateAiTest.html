10  import org.junit.jupiter.api.DisplayName;

15  @DisplayName("LegalAgePredicate tests")
16  class LegalAgePredicateAiTest {
17  
18      private static Stream<Arguments> legalAgePredicateTestCases() {
19          return Stream.of(
20              Arguments.of("User is exactly 18 years old", "2004-03-01", "2022-03-01", true),
21              Arguments.of("User is one day older than 18", "2004-02-28", "2022-03-01", true),
22              Arguments.of("User is one day younger than 18", "2004-03-02", "2022-03-01", false),
23              Arguments.of("User is exactly 19 years old", "2003-03-01", "2022-03-01", true),
24              Arguments.of("User is born today (edge case)", "2022-03-01", "2022-03-01", false)
25          );
26      }
27  
28      @DisplayName("LegalAgePredicate test cases")
29      @ParameterizedTest(name = "[{index}] {0}")
30      @MethodSource("legalAgePredicateTestCases")
31      void legalAgePredicateTest(String description, String dateOfBirthStr, String currentDateStr, boolean expectedResult) {
32          var dateOfBirth = LocalDate.parse(dateOfBirthStr);
33          var currentDate = LocalDate.parse(currentDateStr);
34          var clock = Clock.fixed(currentDate.atStartOfDay(ZoneId.systemDefault()).toInstant(), ZoneId.systemDefault());
35          var legalAgePredicate = new LegalAgePredicate(clock);
36  
37          assertThat(legalAgePredicate.test(dateOfBirth))
38              .as("Check if user with date of birth %s is over 18 on %s", dateOfBirth, currentDate)
39              .isEqualTo(expectedResult);
40      }
41  }
