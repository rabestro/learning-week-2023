1   package epam.flipflop;
2   
3   import java.util.function.Predicate;
4   
5   /**
6    * A predicate that emulates flip-flop logic similar to the two-dots flip-flop in Perl or Ruby.
7    * The predicate takes two input predicates, a start condition, and an end condition.
8    * When the start condition is met, the predicate returns true for all subsequent inputs
9    * until the end condition is met, at which point it returns true and resets its internal state.
10   *
11   * @param <T> The type of the input to the predicate
12   */
13  public class FlipFlopPredicate<T> implements Predicate<T> {
14      private final Predicate<T> startCondition;
15      private final Predicate<T> endCondition;
16      private boolean active;
17  
18      /**
19       * Constructs a new FlipFlopPredicate with the given start and end conditions.
20       *
21       * @param startCondition The predicate that represents the start condition
22       * @param endCondition   The predicate that represents the end condition
23       */
24      public FlipFlopPredicate(Predicate<T> startCondition, Predicate<T> endCondition) {
25          this.startCondition = startCondition;
26          this.endCondition = endCondition;
27          this.active = false;
28      }
29  
30      /**
31       * Evaluates this predicate on the given argument.
32       *
33       * @param t The input argument
34       * @return {@code true} if the input matches the flip-flop logic, otherwise {@code false}
35       */
36      @Override
37      public boolean test(T t) {
38          if (active) {
39              if (endCondition.test(t)) {
40                  active = false;
41              }
42              return true;
43          } else if (startCondition.test(t)) {
44              active = true;
45              return test(t);
46          }
47          return false;
48      }
49  }
