9   @DisplayName("RomanToArabicConverter Test")
10  class RomanToArabicConverterAiTest {
11  
12      private final RomanToArabicConverter converter = new RomanToArabicConverter();
13  
14      @DisplayName("Roman numeral to integer conversion:")
15      @ParameterizedTest(name = "[{index}] {0}: {1} -> {2}")
16      @CsvSource(delimiter = '|', textBlock = """
17              Minimum value                   | I        | 1
18              Only one type of symbol         | III      | 3
19              Subtraction case (IV)           | IV       | 4
20              Subtraction case (IX)           | IX       | 9
21              Subtraction case (XL)           | XL       | 40
22              Subtraction case (XC)           | XC       | 90
23              Subtraction case (CD)           | CD       | 400
24              Subtraction case (CM)           | CM       | 900
25              Combination of symbols          | LVIII    | 58
26              Multiple subtraction cases      | MCMXCIV  | 1994
27              Maximum value                   | MMMCMXCIX| 3999
28              All symbols in descending order | MDCLXVI  | 1666
29          """)
30      void romanToIntegerConversionTest(String description, String input, int expectedOutput) {
31          int result = converter.applyAsInt(input);
32          assertThat(result)
33              .as("Roman numeral %s should be converted to %d", input, expectedOutput)
34              .isEqualTo(expectedOutput);
35      }
36  }
