11  import static org.assertj.core.api.Assertions.assertThat;

13  @DisplayName("FlipFlopPredicate Test")
14  class FlipFlopPredicateAiTest {
15  
16      static Stream<TestCase> testCases() {
17          return Stream.of(
18              new TestCase("Single input, doesn't meet start condition",
19                  List.of(1), x -> x == 2, x -> x == 3, List.of(false)),
20              new TestCase("Single input, meets start condition but not end condition",
21                  List.of(2), x -> x == 2, x -> x == 3, List.of(true)),
22              new TestCase("Single input, meets both start and end conditions",
23                  List.of(2), x -> x == 2, x -> x == 2, List.of(true)),
24              new TestCase("Multiple inputs, start condition met, no end condition met",
25                  List.of(1, 2, 4), x -> x == 2, x -> x == 3, List.of(false, true, true)),
26              new TestCase("Multiple inputs, start condition met, end condition met",
27                  List.of(1, 2, 3, 4), x -> x == 2, x -> x == 3, List.of(false, true, true, false)),
28              new TestCase("Multiple inputs, start and end conditions met multiple times",
29                  List.of(1, 2, 3, 4, 2, 4, 3), x -> x == 2, x -> x == 3,
30                  List.of(false, true, true, false, true, true, true))
31          );
32      }
33  
34      @DisplayName("Test FlipFlopPredicate")
35      @ParameterizedTest(name = "[{index}] {0}")
36      @MethodSource("testCases")
37      void testFlipFlopPredicate(TestCase testCase) {
38          var flipFlopPredicate = new FlipFlopPredicate<>(testCase.startCondition, testCase.endCondition);
39          var results = testCase.inputs.stream().map(flipFlopPredicate::test).toList();
40  
41          assertThat(results)
42              .as("Expected results for the given inputs")
43              .isEqualTo(testCase.expectedResults);
44      }
45  
46      record TestCase(
47          String description,
48          List<Integer> inputs,
49          Predicate<Integer> startCondition,
50          Predicate<Integer> endCondition,
51          List<Boolean> expectedResults
52      ) {
53          @Override
54          public String toString() {
55              return description;
56          }
57      }
58  }
